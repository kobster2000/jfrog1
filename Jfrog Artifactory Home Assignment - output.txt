# JFrog CLI (jf)
# Prerequisites
# install JFrog CLI (jf)
curl -fL https://getcli.jfrog.io | sh
# Move it to your path so you can use 'jf' everywhere
chmod +x jfrog
sudo mv jfrog /usr/local/bin/jf

# Verify Installation
jf --version

# Configure JFrog CLI with your Artifactory instance credentials
jf c add interview-instance \
    --url=https://interviewkobil.jfrog.io \
    --user=kobster2000@gmail.com \
    --password=Password@123 \
    --interactive=false

# Connection Validation
jf rt ping
# Expected result: OK
# CLI constructs the following path: https://interviewkobil.jfrog.io/artifactory/api/system/ping

# Infrastructure Validation
# Display the configured server connections (profiles) stored in your CLI
jf config show
jf config show interview-instance

# List repos
curl -u kobster2000@gmail.com:Password@123 -X GET "https://interviewkobil.jfrog.io/artifactory/api/repositories"
# or via jf CLI
jf rt curl api/repositories
